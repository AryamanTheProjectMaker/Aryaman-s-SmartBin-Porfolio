C++ Arduino Code

cpp
#include <SoftwareSerial.h>
#include <Servo.h>
#include "VoiceRecognitionV3.h"

SoftwareSerial mySerial(2, 3); // RX, TX
VR myVR(mySerial); // Voice Recognition object
Servo lidServo;

uint8_t records[7]; // save record
uint8_t buf[64];

#define open_bin_cmd 0  // Command ID for "open bin"

void setup() {
  Serial.begin(9600);
  mySerial.begin(9600);
  lidServo.attach(9);
  lidServo.write(0); // Lid closed

  if (myVR.begin()) {
    Serial.println("Voice Module Ready.");
  } else {
    Serial.println("Voice Module not detected.");
    while (1);
  }

  if (myVR.load((uint8_t)0, 1) >= 0) {
    Serial.println("Command loaded.");
  } else {
    Serial.println("Command loading failed.");
    while (1);
  }
}

void loop() {
  int ret = myVR.recognize(buf, 50);
  if (ret > 0) {
    switch (buf[1]) {
      case open_bin_cmd:
        Serial.println("Command: Open Bin");
        lidServo.write(90); // Open the lid
        delay(5000);         // Keep open for 5 seconds
        lidServo.write(0);   // Close the lid
        break;
      default:
        Serial.println("Unknown command");
        break;
    }
  }
}

cpp
#include <Servo.h>
#include <SoftwareSerial.h>
#include "VoiceRecognitionV3.h"

// --- Voice Recognition Setup ---
SoftwareSerial mySerial(3, 4); // RX, TX for voice module
VR myVR(mySerial); // Voice Recognition object
#define open_bin_cmd 0  // ID for "open bin"

bool open = false;
unsigned long time = 0;

const int trigPin = 12;
const int echoPin = 11;
const int servoPin = 6;
int blueLED = 2;
int a = 10;
int redButton = 8;
int greenButton = 9;
int blueButton = 10;

long duration;
float distance;

Servo myServo;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(blueLED, OUTPUT);
  pinMode(redButton, INPUT_PULLUP);
  pinMode(greenButton, INPUT_PULLUP);
  pinMode(blueButton, INPUT_PULLUP);

  myServo.attach(servoPin);
  myServo.write(0);

  digitalWrite(trigPin, LOW);
  delay(100);

  Serial.begin(9600);
  mySerial.begin(9600);

  if (myVR.begin()) {
    Serial.println("Voice Module Ready.");
  } else {
    Serial.println("Voice Module not detected.");
    while (1);
  }

  if (myVR.load((uint8_t)0, 1) >= 0) {
    Serial.println("Voice command loaded.");
  } else {
    Serial.println("Failed to load voice command.");
    while (1);
  }
}

void loop() {
  // Handle mode switching with buttons
  if (digitalRead(redButton) == LOW) {
    a = 10;
  }
  if (digitalRead(greenButton) == LOW) {
    a = 20;
  }
  if (digitalRead(blueButton) == LOW) {
    a = 30;
  }

  // Ultrasonic sensor reading
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = (duration / 2.0) / 29.1;

  Serial.print("Distance: ");
  Serial.println(distance);
  Serial.print("A = ");
  Serial.println(a);

  // Check for voice command
  int ret = myVR.recognize(buf, 50);
  if (ret > 0) {
    if (buf[1] == open_bin_cmd) {
      Serial.println("Voice Command: Open Bin");
      time = millis();
      open = true;
      myServo.write(90);
      digitalWrite(blueLED, HIGH);
    }
  }

  // Ultrasonic auto-open
  if (distance > 0 && distance <= a) {
    time = millis();
    open = true;
    myServo.write(90);
    digitalWrite(blueLED, HIGH);
  }

  // Auto-close after 4 seconds
  if ((open && millis() - time > 4000) || !open) {
    if (distance > a) {
      myServo.write(0);
      digitalWrite(blueLED, LOW);
      open = false;
    }
  }

  delay(100);
}
